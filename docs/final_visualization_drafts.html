Read the NBA data

nba_data <- read.csv("nba_data_processed.csv", stringsAsFactors = FALSE)

Filter for qualified players

qualified_players <- subset(nba_data, G >= 50 & MP >= 20)

Calculate additional advanced metrics

qualified_players$TS_Pct <- qualified_players$PTS / (2 * (qualified_players$FGA + 0.44 * qualified_players$FTA)) * 100
qualified_players$USG_Pct <- (qualified_players$FGA + 0.44 * qualified_players$FTA + qualified_players$TOV) / (qualified_players$MP / 5) * 100
qualified_players$BPM_approx <- (qualified_players$PTS + 2.2 * qualified_players$STL + 0.7 * qualified_players$BLK + 
                                 0.3 * qualified_players$AST + 0.7 * qualified_players$DRB + 
                                 0.3 * qualified_players$ORB - 0.6 * qualified_players$FGA - 
                                 0.6 * qualified_players$FTA - 0.4 * qualified_players$PF - 
                                 qualified_players$TOV) / qualified_players$G
qualified_players$VORP_approx <- qualified_players$BPM_approx * (qualified_players$MP / (48 * 82)) * 2.7
qualified_players$WS_approx <- (qualified_players$PTS * 0.037 + qualified_players$STL * 0.138 + 
                               qualified_players$BLK * 0.095 + qualified_players$ORB * 0.034 + 
                               qualified_players$DRB * 0.034 + qualified_players$AST * 0.048 - 
                               qualified_players$FGA * 0.029 - qualified_players$FTA * 0.015 - 
                               qualified_players$TOV * 0.034 - qualified_players$PF * 0.020) * qualified_players$G
qualified_players$PER_approx <- (qualified_players$PTS + 2.7 * qualified_players$STL + 
                                0.7 * qualified_players$BLK + 0.7 * qualified_players$AST + 
                                0.3 * qualified_players$ORB + 0.3 * qualified_players$DRB - 
                                0.7 * qualified_players$TOV - 0.4 * qualified_players$PF) / qualified_players$MP * 36
qualified_players$ORtg_approx <- (qualified_players$PTS / (qualified_players$FGA + 
                                 0.44 * qualified_players$FTA + qualified_players$TOV)) * 100
qualified_players$DRtg_approx <- 110 - (qualified_players$STL * 0.5 + qualified_players$BLK * 0.5 + 
                                       qualified_players$DRB * 0.1) * 100 / qualified_players$MP

VISUALIZATION 3: Win Shares by Position

simplify_position <- function(pos) {
  if (grepl("PG", pos)) return("PG")
  if (grepl("SG", pos)) return("SG")
  if (grepl("SF", pos)) return("SF")
  if (grepl("PF", pos)) return("PF")
  if (grepl("C", pos)) return("C")
  return(pos)  # Default case
}

qualified_players$SimplifiedPos <- sapply(qualified_players$Pos, simplify_position)

viz3 <- ggplot(qualified_players, aes(x = SimplifiedPos, y = WS_approx, fill = SimplifiedPos)) +

VISUALIZATION 7: Offensive and Defensive Contribution Radar Chart

sorted_per <- qualified_players[order(-qualified_players$PER_approx), ]
top_players <- sorted_per[1:5, ]

radar_stats <- c("PTS", "AST", "STL", "BLK", "TRB", "eFG.")
radar_data <- top_players[, c("Player", radar_stats)]

for(stat in radar_stats) {
  max_val <- max(qualified_players[[stat]], na.rm = TRUE)
  radar_data[[stat]] <- radar_data[[stat]] / max_val
}

radar_long <- reshape2::melt(radar_data, id.vars = "Player", 
                            variable.name = "Stat", value.name = "Value")

viz7 <- ggplot(radar_long, aes(x = Stat, y = Value, group = Player, color = Player)) +
